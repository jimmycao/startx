#-----------------------------------------------------
# install
#-----------------------------------------------------
by default, git clone startx to ~/startx, then

1) add '. ~/startx/my_bashrc' in ~/.bashrc

alias vi='vim'
alias rm='rm -i'
alias cp='cp -i'
alias mv='mv -i'

. ~/startx/my_bashrc

2) ln -s ~/startx/hidden_vim/hidden_vimrc ~/.vimrc
3) ln -s ~/startx/hidden_vim ~/.vim



#-----------------------------------------------------
# Trouble-Shooting
#-----------------------------------------------------
1) check your vim version, update it if necessary
